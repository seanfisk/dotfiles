# Powerline stuff
#
# Powerline is able to work with pygit2 -or- the plain git executable.
# Installing pygit2 is disabled by default because it has dependencies
# (libgit2, which depends on cmake) which are not necessarily simple
# to install. That said, with Homebrew, installing pygit2 is rather
# easy so we may as well. Assuming your system Python is
# '/usr/bin/python' and that pip is installed to the user site, run:
#
#     brew install libgit2
#
# pygit2 requires cffi to be installed first (annoyingly):
#
#     system_python_pip="$(/usr/bin/python -m site --user-base)/bin/pip"
#     "$system_python_pip" install --user cffi==0.8.6
#
# Finally install pygit2:
#
#     "$system_python_pip" install --user pygit2==0.21.2
#
powerline-status==1.0
psutil==2.1.1
bzr==2.6.0
pycparser==2.10
cffi==0.8.6
mercurial==3.1

# Portable archive file manager
